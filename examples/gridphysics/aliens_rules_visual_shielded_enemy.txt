# Aliens visual rule variant: shielded_enemy. Player shots turn enemies into different sprites before defeat.
BasicGame square_size=32
    SpriteSet
        background > Immovable img=oryx/space1 hidden=True
        base    > Immovable    color=WHITE img=oryx/planet
        avatar  > FlakAvatar   stype=sam img=oryx/spaceship1
        missile > Missile
            sam  > orientation=UP    color=BLUE singleton=True img=oryx/bullet1
            bomb > orientation=DOWN  color=RED  speed=0.5 img=oryx/bullet2
        alien   > Bomber       stype=bomb   prob=0.01  cooldown=3 speed=0.8
            alienGreen > img=oryx/alien3
            alienBlue > img=oryx/alien1
        visualAlien > Bomber stype=bomb prob=0.01 cooldown=3 speed=0.8
            visualAlienShielded > img=oryx/ghost3
            visualAlienExposed > img=oryx/ghost1
        portal  > invisible=True hidden=True
        	portalSlow  > SpawnPoint   stype=alienBlue  cooldown=16   total=20 img=portal
        	portalFast  > SpawnPoint   stype=alienGreen  cooldown=12   total=20 img=portal

    LevelMapping
        . > background
        0 > background base
        1 > background portalSlow
        2 > background portalFast
        A > background avatar

    TerminationSet
        SpriteCounter      stype=avatar               limit=0 win=False
        MultiSpriteCounter stype1=portal stype2=alien stype3=visualAlien limit=0 win=True

    InteractionSet
        avatar  EOS  > stepBack
        alien   EOS  > turnAround
        visualAlien EOS > turnAround
        missile EOS  > killSprite

        base bomb > killBoth
        base sam > killBoth scoreChange=1

        base   alien > killSprite
        base   visualAlien > killSprite
        avatar alien > killSprite scoreChange=-1
        avatar visualAlien > killSprite scoreChange=-1
        avatar bomb  > killSprite scoreChange=-1
        visualAlienExposed sam > killSprite scoreChange=2
        visualAlienShielded sam > transformTo stype=visualAlienExposed killSecond=True forceOrientation=True
        alienGreen sam > transformTo stype=visualAlienShielded killSecond=True forceOrientation=True
        alienBlue sam > transformTo stype=visualAlienShielded killSecond=True forceOrientation=True
